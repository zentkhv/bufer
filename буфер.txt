import subprocess

try:
    output = subprocess.check_output(['pgrep', '-f', 'get_metrics.py'])
    if output.strip():
        print("Процесс 'get_metrics.py' найден")
    else:
        print("Процесс 'get_metrics.py' не найден")
except subprocess.CalledProcessError:
    print("Произошла ошибка при выполнении команды 'pgrep'.")
