# Запрашиваем у пользователя имена файлов
input_filename = input("Введите имя входного файла: ")
output_filename = input("Введите имя выходного файла: ")

# Открываем входной файл для чтения
with open(input_filename, 'r') as file:
    contents = file.readlines() # читаем содержимое файла построчно

# Создаем список для хранения названий групп
group_names = []

# Проходим по содержимому файла
for line in contents:
    line = line.strip() # удаляем пробелы и переносы строк
    if line.startswith("[") and line.endswith("]"): # если строка начинается и заканчивается на скобки
        group_name = line[1:-1] # извлекаем название группы, удаляя скобки
        group_names.append(group_name) # добавляем название группы в список

# Открываем выходной файл для записи
with open(output_filename, 'w') as file:
    for group_name in group_names:
        file.write(group_name + "\n") # записываем название группы в выходной файл

print("Названия групп записаны в файл " + output_filename)
print("Количество записанных групп: " + str(len(group_names))) # выводим количество записанных групп
